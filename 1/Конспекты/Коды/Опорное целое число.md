``` python
class Solution:
    def pivotInteger(self, n: int) -> int:
        res0 = (n * (n + 1) // 2)   # формула Гаусса для нахождения общей суммы
        res1 = int(res0**0.5)   # находим квдратный корень от общей суммы

        if res1 * res1 == res0:   # если произведение корней равно общей сумме, вернуть корень
            return res1
        return -1   # если условие не сработает, вернётся -1
```
#### формула Гаусса
Используем её для нахождения общей суммы
#### Квадратный корень
Находим квадратный корень от общей суммы 
#### Проверяем
Равно ли произведение корней общей сумме
Если нет возвращаем значение -1
